## Install

```bash
npm install --save pooler-sdk
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'pooler-sdk'
import 'pooler-sdk/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
